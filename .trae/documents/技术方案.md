## 三、模块一：实时对话引擎（客户端直连 Deepgram）

### 3.1 核心设计：客户端直连

**关键决策：ASR 不走后端，客户端直接 WebSocket 连接 Deepgram。**
好处：

- 省掉整个 WebSocket Gateway 服务端
- 延迟更低（少一跳）
- 后端完全 Serverless 化
  安全性：
- Deepgram API Key 不能暴露给客户端
- 解决方案：通过 Supabase Edge Function 生成短期 token

```typescript
// Edge Function: /functions/v1/deepgram-token
// 生成 Deepgram 临时 API Key（有效期 10 分钟）
import { serve } from  https://deno.land/std@0.168.0/http/server.ts ;
import { createClient } from  https://esm.sh/@supabase/supabase-js@2 ;

serve(async (req) => {
  // 1. 验证用户身份
  const supabase = createClient(
    Deno.env.get( SUPABASE_URL )!,
    Deno.env.get( SUPABASE_ANON_KEY )!,
    { global: { headers: { Authorization: req.headers.get( Authorization )! } } }
  );
  const { data: { user }, error } = await supabase.auth.getUser();
  if (error || !user) return new Response( Unauthorized , { status: 401 });

  // 2. 检查用量限制（Free 用户每天 15 分钟）
  const usage = await checkDailyUsage(supabase, user.id);
  if (usage.exceeded) {
    return Response.json({ error:  Daily limit exceeded  }, { status: 429 });
  }

  // 3. 向 Deepgram 申请临时 Key
  const resp = await fetch( https://api.deepgram.com/v1/keys/  + Deno.env.get( DG_PROJECT_ID ), {
    method:  POST ,
    headers: {
       Authorization :  Token   + Deno.env.get( DEEPGRAM_API_KEY ),
       Content-Type :  application/json ,
    },
    body: JSON.stringify({
      comment: `temp-key-${user.id}`,
      scopes: [ usage:write ],
      time_to_live_in_seconds: 600, // 10 分钟有效
    }),
  });

  const { key } = await resp.json();
  return Response.json({ deepgram_token: key, expires_in: 600 });
});
```

### 3.2 客户端音频采集与 ASR

```typescript
// AudioEngine.ts — 客户端核心模块
import { Audio } from "expo-av";

class AudioEngine {
  private dgSocket: WebSocket | null = null;
  private onTranscript: (
    speaker: string,
    text: string,
    isFinal: boolean,
  ) => void;
  private onUtteranceEnd: (speaker: string, lastText: string) => void;

  /**
   * 建立双通道 Deepgram 连接
   * Channel A: 耳机麦克风 → self
   * Channel B: 手机麦克风 → other
   *
   * MVP 简化：先用单通道 + 声纹区分的降级方案
   * 因为 Expo 的双通道采集需要 native module
   */
  async startSession(token: string) {
    // 连接 Deepgram Streaming API
    const dgUrl =
      `wss://api.deepgram.com/v1/listen?` +
      `model=nova-2` +
      `&language=en` +
      `&smart_format=true` +
      `&interim_results=true` +
      `&utterance_end_ms=1500` +
      `&vad_events=true` +
      `&punctuate=true` +
      `&diarize=true` + // 开启 Deepgram 内置说话人分离
      `&encoding=linear16` +
      `&sample_rate=16000` +
      `&channels=1`;

    this.dgSocket = new WebSocket(dgUrl, ["token", token]);

    this.dgSocket.onmessage = (event) => {
      const data = JSON.parse(event.data);

      if (data.type === "Results") {
        const alt = data.channel.alternatives[0];
        const words = alt.words || [];
        // Deepgram diarize 返回每个 word 的 speaker 标签
        const speaker = this.determineSpeaker(words);
        this.onTranscript(speaker, alt.transcript, data.is_final);
      }

      if (data.type === "UtteranceEnd") {
        // 触发建议生成或 Review
        this.onUtteranceEnd(this.lastSpeaker, this.lastFinalTranscript);
      }
    };

    // 开始录音并流式发送
    await this.startRecording();
  }

  private determineSpeaker(words: any[]): "self" | "other" {
    // Deepgram diarize 给每个 word 标 speaker: 0 或 1
    // 首次使用时需要用户标定：哪个 speaker ID 是自己
    // 标定方法：让用户先说一句  Hello, this is me
    if (!words.length) return "other";
    const speakerId = words[0].speaker;
    return speakerId === this.selfSpeakerId ? "self" : "other";
  }

  private async startRecording() {
    const recording = new Audio.Recording();
    await recording.prepareToRecordAsync({
      android: {
        extension: ".wav",
        outputFormat: Audio.AndroidOutputFormat.DEFAULT,
        audioEncoder: Audio.AndroidAudioEncoder.DEFAULT,
        sampleRate: 16000,
        numberOfChannels: 1,
        bitRate: 256000,
      },
      ios: {
        extension: ".wav",
        audioQuality: Audio.IOSAudioQuality.HIGH,
        sampleRate: 16000,
        numberOfChannels: 1,
        bitRate: 256000,
        linearPCMBitDepth: 16,
        linearPCMIsBigEndian: false,
        linearPCMIsFloat: false,
      },
    });

    // 使用 onRecordingStatusUpdate 定期获取音频数据并发送
    recording.setOnRecordingStatusUpdate(async (status) => {
      if (status.isRecording && this.dgSocket?.readyState === WebSocket.OPEN) {
        // 读取最新音频 chunk 并发送到 Deepgram
        const chunk = await this.getLatestAudioChunk(recording);
        if (chunk) {
          this.dgSocket.send(chunk);
        }
      }
    });

    await recording.startAsync();
  }
}
```

### 3.3 说话人分离方案（轻量版）

**MVP 方案：Deepgram 内置 Diarization + 用户标定**


### 3.4 端侧 ASR 方案：Whisper.cpp（Free 用户零成本）

#### 3.4.1 方案概述

**核心思路：Free 用户使用端侧 Whisper 模型进行语音识别，ASR 成本直接归零。Pro 用户使用云端 Groq Whisper API 获得更高质量体验。**
这是降低 ASR 成本最关键的架构决策——ASR 占总变动成本的 53%，端侧化后 Free 用户的 ASR 成本从每月数千美元降为 $0。

#### 3.4.3 Whisper.cpp 端侧集成

**模型选择**：

| 模型        | 大小      | iPhone 12+ 延迟 | WER      | 推荐         |
| ----------- | --------- | --------------- | -------- | ------------ |
| tiny.en     | 75MB      | ~200ms/段       | ~15%     | 低端设备降级 |
| **base.en** | **142MB** | **~400ms/段**   | **~12%** | **MVP 首选** |
| small.en    | 466MB     | ~1.2s/段        | ~9%      | 后续可选升级 |

**React Native 集成（使用 whisper.rn 库）**：

```typescript
// WhisperEngine.ts — 端侧 ASR 引擎
import { initWhisper, WhisperContext } from "whisper.rn";

class WhisperEngine {
  private whisperCtx: WhisperContext | null = null;
  private modelReady = false;

  /**
   * 初始化 Whisper 模型
   * 首次使用时下载模型文件（142MB），后续从本地缓存加载
   */
  async initialize() {
    const modelPath = await this.ensureModelDownloaded("ggml-base.en.bin");

    this.whisperCtx = await initWhisper({
      filePath: modelPath,
      // 启用 CoreML 加速 (iOS) / NNAPI (Android)
      useCoreMLIos: true,
      useGpu: true,
    });

    this.modelReady = true;
  }

  /**
   * VAD 检测到一段语音结束后，调用端侧 Whisper 转录
   */
  async transcribe(audioFilePath: string): Promise<TranscribeResult> {
    if (!this.whisperCtx) throw new Error("Whisper not initialized");

    const startTime = Date.now();

    const result = await this.whisperCtx.transcribe(audioFilePath, {
      language: "en",
      maxLen: 0, // 不限段长
      translate: false,
      beamSize: 1, // beam=1 最快，质量损失小
      bestOf: 1,
    });

    const latency = Date.now() - startTime;
    console.log(`Whisper transcribe: ${latency}ms`);

    return {
      text: result.data
        .map((seg) => seg.text)
        .join(" ")
        .trim(),
      segments: result.data,
      latency_ms: latency,
    };
  }

  /**
   * 模型下载与缓存管理
   */
  private async ensureModelDownloaded(modelName: string): Promise<string> {
    const localPath = `${FileSystem.documentDirectory}models/${modelName}`;
    const exists = await FileSystem.getInfoAsync(localPath);

    if (!exists.exists) {
      // 首次下载，展示进度条
      const downloadUrl = `https://huggingface.co/ggerganov/whisper.cpp/resolve/main/${modelName}`;
      await FileSystem.downloadAsync(downloadUrl, localPath, {
        onProgress: (progress) => {
          useAppStore
            .getState()
            .setModelDownloadProgress(
              progress.totalBytesWritten / progress.totalBytesExpectedToWrite,
            );
        },
      });
    }

    return localPath;
  }
}
```

#### 3.4.4 VAD 切段 + Whisper 的完整流程

端侧 Whisper 不支持流式识别，需要配合 VAD（Voice Activity Detection）进行 说完一句就识别 ：

```plaintext
[持续录音] → [端侧 VAD 检测语音活动]
                ↓ 检测到停顿 > 1.5s
            [截取该段音频]
                ↓
            [送入 Whisper.cpp 端侧推理]
                ↓ ~400ms
            [返回文字结果]
                ↓
            [更新 UI + 触发建议/Review]

```

```typescript
// VADManager.ts — 端侧 VAD + Whisper 组合
import { useSileroVAD } from "@ricky0123/vad-react-native";

class VADManager {
  private whisperEngine: WhisperEngine;
  private audioRecorder: AudioRecorder;

  /**
   * 使用 Silero VAD 检测语音段落
   * Silero VAD 模型仅 ~2MB，端侧运行，零成本
   */
  startListening() {
    useSileroVAD({
      onSpeechStart: () => {
        // 开始录制当前语音段
        this.audioRecorder.startSegment();
      },
      onSpeechEnd: async (audioData: Float32Array) => {
        // 语音段结束，保存为文件
        const segmentPath = await this.audioRecorder.saveSegment(audioData);

        // 端侧 Whisper 转录
        const result = await this.whisperEngine.transcribe(segmentPath);

        if (result.text.trim().length > 0) {
          // 推送到 UI
          useConversationStore.getState().addTranscript({
            text: result.text,
            timestamp: Date.now(),
            source: "on-device",
          });
        }

        // 清理临时音频文件
        await FileSystem.deleteAsync(segmentPath);
      },
      // VAD 参数
      positiveSpeechThreshold: 0.8,
      negativeSpeechThreshold: 0.3,
      minSpeechFrames: 5,
      preSpeechPadFrames: 10,
    });
  }
}
```

#### 3.4.5 端侧方案的说话人分离

端侧 Whisper 本身不支持 diarize，需要额外方案：

| 方案                         | 说明                                                   | 推荐度   |
| ---------------------------- | ------------------------------------------------------ | -------- |
| **用户标定 + 音量/方向推断** | 利用麦克风方向和音量差异（自己说话声音更大）做粗略区分 | MVP 可用 |
| 耳机物理分离                 | 耳机麦克风=自己，手机麦克风=对方                       | 最准确   |
| 端侧声纹（sherpa-onnx）      | 轻量端侧声纹模型，~5MB                                 | 后续升级 |

MVP 阶段推荐最简方案：

```typescript
// 简易说话人区分：基于音量阈值
function classifySpeaker(audioData: Float32Array): "self" | "other" {
  const rms = Math.sqrt(
    audioData.reduce((sum, x) => sum + x * x, 0) / audioData.length,
  );
  // 自己的声音离麦克风近，RMS 通常更高
  // 阈值在首次标定时确定
  return rms > selfVolumeThreshold ? "self" : "other";
}
```

```typescript
        Authorization: `Bearer ${accessToken}`,
      },
```

### 4.2 Edge Function：生成建议

```typescript
const openai = new OpenAI({ apiKey: Deno.env.get( OPENAI_API_KEY ) });

    .reverse()

  // 2. 构建 Prompt
  const systemPrompt = `You are a real-time English conversation assistant helping a non-native speaker.
Generate 2 reply suggestions in different styles based on the conversation context.

Rules:
-  formal : professional and polite
-  casual : natural everyday English
- Keep each suggestion to 1-3 sentences
- Be contextually appropriate
- Output valid JSON only

Output format:
{ suggestions :[{ style : formal , text : ... },{ style : casual , text : ... }]}`;

  const userPrompt = `Scene: ${scene ||  General conversation }

Recent conversation:
${conversationContext}

The other person just said:  ${last_utterance}

Generate 2 reply suggestions:`;

  // 3. 调用 OpenAI Streaming
  const stream = await openai.chat.completions.create({
    model:  gpt-4o-mini ,
    messages: [
      { role:  system , content: systemPrompt },
      { role:  user , content: userPrompt },
    ],
    max_tokens: 250,
    temperature: 0.7,
    stream: true,
  });

  // 4. 流式返回给客户端
  const encoder = new TextEncoder();
  const readable = new ReadableStream({
    async start(controller) {
      for await (const chunk of stream) {
        const content = chunk.choices[0]?.delta?.content ||   ;
        if (content) {
          controller.enqueue(encoder.encode(`data: ${content}\n\n`));
        }
      }
      controller.enqueue(encoder.encode( data: [DONE]\n\n ));
      controller.close();
    },
  });

  return new Response(readable, {
    headers: {
       Content-Type :  text/event-stream ,
       Cache-Control :  no-cache ,
       Connection :  keep-alive ,
    },
  });
});
```

```typescript
// SuggestionService.ts
class SuggestionService {
  async fetchSuggestions(
    sessionId: string,
    lastUtterance: string,
    scene: string
  ): Promise<void> {
    const { data: { session } } = await supabase.auth.getSession();

    const response = await fetch(
      `${SUPABASE_URL}/functions/v1/suggest`,
      {
        method:  POST ,
        headers: {
           Authorization : `Bearer ${session?.access_token}`,
           Content-Type :  application/json ,
        },
        body: JSON.stringify({
          session_id: sessionId,
          last_utterance: lastUtterance,
          scene: scene,
        }),
      }
    );

    // 流式读取
    const reader = response.body!.getReader();
    const decoder = new TextDecoder();
    let buffer =   ;

    while (true) {
      const { done, value } = await reader.read();
      if (done) break;

      buffer += decoder.decode(value, { stream: true });
      const lines = buffer.split( \n\n );
      buffer = lines.pop() ||   ;

      for (const line of lines) {
        const data = line.replace( data:  ,   );
        if (data ===  [DONE] ) break;
        // 增量追加到 UI 状态
        useSuggestionStore.getState().appendChunk(data);
      }
    }

    // 解析完整 JSON 并展示
    useSuggestionStore.getState().finalizeSuggestions();
  }
}
```

## 五、模块三：即时语言 Review

### 5.1 Edge Function：Review

```typescript
// Edge Function: /functions/v1/review
serve(async (req) => {
  const supabase = createClient(
    Deno.env.get( SUPABASE_URL )!,
    Deno.env.get( SUPABASE_SERVICE_ROLE_KEY )!
  );

  const { session_id, user_utterance, scene } = await req.json();

  // 跳过太短的句子
  if (user_utterance.split(   ).length < 4) {
    return Response.json({ overall_score: "green", issues: [], skipped: true });
  }

  // 获取上下文
  const { data: recentTurns } = await supabase
    .from("turns")
    .select("speaker, text")
    .eq("session_id", session_id)
    .order("created_at", { ascending: false })
    .limit(6);

  const context = (recentTurns || [])
    .reverse()
    .map((t) => `${t.speaker ===  other  ?  Other  :  You }: ${t.text}`)
    .join( \n );

  const systemPrompt = `You are an English language coach reviewing a non-native speaker's utterance during a live conversation.

Rules:
- Focus on max 2 most impactful issues
- Spoken English is more relaxed than written — only flag real errors
- Provide corrections with brief Chinese explanations
- Be encouraging

Scoring:
-  green : natural, no issues
-  yellow : minor issues or more natural alternatives exist
-  red : clear grammar errors or confusing expression

Output JSON only:
{
   overall_score :  green|yellow|red ,
   issues : [{ type : grammar|vocabulary , original : ... , corrected : ... , explanation : 中文简短解释 }],
   better_expression :  more natural way to say it (or null) ,
   praise :  positive feedback if applicable (or null)
}`;

  const userPrompt = `Context:\n${context}\n\nScene: ${scene ||  General }\nUser said:  ${user_utterance} \n\nReview:`;

  // 非流式调用（Review 需要完整 JSON）
  const completion = await openai.chat.completions.create({
    model:  gpt-4o-mini ,
    messages: [
      { role:  system , content: systemPrompt },
      { role:  user , content: userPrompt },
    ],
    max_tokens: 200,
    temperature: 0.3,
    response_format: { type:  json_object  },
  });

  const review = JSON.parse(completion.choices[0].message.content!);

  // 异步存储 review 结果（不阻塞返回）
  supabase.from( reviews ).insert({
    session_id,
    user_utterance,
    overall_score: review.overall_score,
    issues: review.issues,
    better_expression: review.better_expression,
  }).then(); // fire-and-forget

  return Response.json(review);
});
```

```sql
-- Edge Function 中调用此函数检查用量
create or replace function check_daily_usage(p_user_id uuid)
returns json as $$
declare
  v_profile profiles%rowtype;
  v_limit integer;
begin
  select * into v_profile from profiles where id = p_user_id;

  -- 日期翻转则重置计数
  if v_profile.daily_minutes_reset_at < current_date then
    update profiles
    set daily_minutes_used = 0, daily_minutes_reset_at = current_date
    where id = p_user_id;
    v_profile.daily_minutes_used := 0;
  end if;

  -- Free 用户 15 分钟，Pro 不限
  v_limit := case v_profile.subscription_tier
    when 'free' then 15
    else 999999
  end;

  return json_build_object(
    'used', v_profile.daily_minutes_used,
    'limit', v_limit,
    'exceeded', v_profile.daily_minutes_used >= v_limit
  );
end;
$$ language plpgsql security definer;
```
